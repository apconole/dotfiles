This is my Emacs configuration written as a literate program.  All code lives
in =#+begin_src emacs-lisp= blocks and is loaded at startup via
=org-babel-load-file= from =init.el=.

Sensitive settings — email accounts, IMAP/SMTP servers, IRC credentials,
local git repository paths — live in =private.org= / =private.el=, which is
*not* tracked in version control.  See the [[*Private Configuration][Private Configuration]] section for
details.

** Package Management

Use MELPA as the package archive.  Two helper functions handle
install-on-demand so that the config is self-bootstrapping on a fresh machine.

#+begin_src emacs-lisp
  (require 'package)
  (mapc (lambda (p) (push p package-archives))
        '(("melpa-stable" . "http://melpa.org/packages/")))
  (package-initialize)
#+end_src

Now that MELPA support is setup, we want to make sure we refresh packages.
If we refreshed every time we started emacs, things would be slow, but we
can setup an interval and check it every time we start.
#+begin_src emacs-lisp
      (defvar simple-emacs-package-refresh-interval (* 7 24 60 60)
        "Number of seconds between automatic package refreshes (1 week).")

      (defvar simple-emacs-last-package-refresh-file
        (expand-file-name "last-package-refresh" user-emacs-directory)
        "File storing the timestamp of the last package refresh.")

      (defun simple-emacs-should-refresh-packages-p ()
        "Return t if packages should be refreshed."
        (let ((last-refresh (when
      			  (file-exists-p simple-emacs-last-package-refresh-file)
                              (string-to-number (with-temp-buffer
                                                  (insert-file-contents
    					       simple-emacs-last-package-refresh-file)
                                                  (buffer-string))))))
          (or (null last-refresh)
              (> (float-time (current-time)) (+ last-refresh
  					      simple-emacs-package-refresh-interval)))))

      (defun simple-emacs-update-last-package-refresh ()
        "Update the timestamp of the last package refresh."
        (with-temp-file simple-emacs-last-package-refresh-file
          (insert (number-to-string (float-time (current-time))))))

      ;; Automatically refresh packages once a week
      (when (simple-emacs-should-refresh-packages-p)
        (package-refresh-contents)
        (simple-emacs-update-last-package-refresh))
#+end_src

These will help to just pull a package if we need it.

** Utility Functions

Helper functions used throughout the rest of the configuration.

#+begin_src emacs-lisp
(defun get-closest-pathname (file)
  "Return the path of FILE in the nearest ancestor directory that contains it.
Walks upward from `default-directory'."
  (let ((root (expand-file-name "/")))
    (expand-file-name
     file
     (cl-loop for d = default-directory then (expand-file-name ".." d)
              if (file-exists-p (expand-file-name file d)) return d
              if (equal d root) return nil))))

(defun upward-find-file (filename &optional startdir)
  "Walk up from STARTDIR (default: \".\") until FILENAME is found.
Returns the containing directory with a trailing slash, or nil."
  (let ((dirname (expand-file-name (or startdir ".")))
        (found nil)
        (top nil))
    (while (not (or found top))
      (when (string= (expand-file-name dirname) "/")
        (setq top t))
      (if (file-exists-p (expand-file-name filename dirname))
          (setq found t)
        (setq dirname (expand-file-name ".." dirname))))
    (when found (concat dirname "/"))))
#+end_src

* Private Configuration

Sensitive configuration (email accounts, IMAP/SMTP servers, IRC credentials,
GPG key IDs, local repository paths) lives in =private.el=, which is tangled
from =private.org=.

To set up on a new machine:
1. Copy =private.org.example= to =private.org= and fill in the real values.
2. Open =private.org= and run =M-x org-babel-tangle= to generate =private.el=.
3. Restart Emacs (or =M-x eval-buffer= the generated file).

=private.org= and =private.el= should be added to =.gitignore= so that
credentials are never committed.

#+begin_src emacs-lisp
(load (expand-file-name "private.el" user-emacs-directory) t t)
#+end_src

** Git Timemachine

#+begin_src emacs-lisp
(defun git-timemachine-other-window ()
  "Open an indirect buffer of the current file and run git-timemachine on it."
  (interactive)
  (let ((buf-name   (generate-new-buffer-name "*Timemachine Mode*"))
        (file-name  (buffer-file-name)))
    (pop-to-buffer (make-indirect-buffer (current-buffer) buf-name))
    (setq buffer-file-name file-name)
    (goto-char (point-min))
    (git-timemachine)
    (local-set-key (kbd "q") 'kill-this-buffer)))

** Narrowing / Widening

Smart =C-x n= that does the right thing depending on context:
- Already narrowed → widens.
- Active region → narrow to region.
- In org-mode on a src block → =org-edit-src-code=.
- In org-mode on a block → narrow to block.
- In org-mode elsewhere → narrow to subtree.
- Otherwise → narrow to defun.

Credit: Michael Fogleman, Artur Malabarba, endlessparentheses.com.

#+begin_src emacs-lisp
(defun narrow-or-widen-dwim (p)
  "Widen if narrowed; otherwise narrow intelligently.
With prefix P, always narrow even if already narrowed."
  (interactive "P")
  (declare (interactive-only))
  (cond ((and (buffer-narrowed-p) (not p)) (widen))
        ((region-active-p)
         (narrow-to-region (region-beginning) (region-end)))
        ((derived-mode-p 'org-mode)
         (cond ((ignore-errors (org-edit-src-code))
                (delete-other-windows))
               ((org-at-block-p)
                (org-narrow-to-block))
               (t (org-narrow-to-subtree))))
        (t (narrow-to-defun))))

(define-key ctl-x-map "n" #'narrow-or-widen-dwim)
(eval-after-load 'org-src
  '(define-key org-src-mode-map "\C-x\C-s" #'org-edit-src-exit))
#+end_src

** w3m Code Search

=M-/= searches Google for the symbol at point (or active region), scoped to
files of the same type as the current buffer.

The search engine list uses short keys for use with =C-u S <key> RET= inside
w3m as well.

| Key | Search target                         |
|-----+---------------------------------------|
| =g= | Google                                |
| =q= | Stack Overflow                        |
| =s= | Filetype-scoped (default: .el)        |
| =w= | Wikipedia                             |
| =d= | dictionary.reference.com             |
| =j= | MDN (JavaScript / Mozilla)           |

#+begin_src emacs-lisp
(simple-emacs-package-install 'w3m)

(defun w3m-get-url-from-search-engine-alist (k l)
  (when (listp l)
    (if (string= k (caar l))
        (nth 1 (car l))
      (w3m-get-url-from-search-engine-alist k (cdr l)))))

(defun w3m-set-url-from-search-engine-alist (k l url)
  (when (listp l)
    (if (string= k (caar l))
        (setcdr (car l) (list url))
      (w3m-set-url-from-search-engine-alist k (cdr l) url))))

(setq w3m-search-engine-alist
      '(("g" "http://www.google.com.au/search?q=%s" utf-8)
        ("q" "http://www.google.com.au/search?q=%s+site:stackoverflow.com" utf-8)
        ("s" "http://www.google.com.au/search?q=%s+filetype:el" utf-8)
        ("w" "http://en.wikipedia.org/wiki/Special:Search?search=%s" utf-8)
        ("d" "http://dictionary.reference.com/search?q=%s" utf-8)
        ("j" "http://www.google.com.au/search?q=%s+site:developer.mozilla.org" utf-8)))

(defun w3m-google-by-filetype ()
  "Search Google for the symbol at point scoped to the current file extension."
  (interactive)
  (unless (featurep 'w3m) (require 'w3m))
  (let* ((thing (if (region-active-p)
                    (buffer-substring-no-properties (region-beginning) (region-end))
                  (thing-at-point 'symbol)))
         (old-url (w3m-get-url-from-search-engine-alist "s" w3m-search-engine-alist)))
    (when buffer-file-name
      (w3m-set-url-from-search-engine-alist
       "s" w3m-search-engine-alist
       (replace-regexp-in-string "filetype:.*"
                                 (concat "filetype:" (file-name-extension buffer-file-name))
                                 old-url)))
    (w3m-search "s" thing)
    (w3m-set-url-from-search-engine-alist "s" w3m-search-engine-alist old-url)))

(global-set-key (kbd "M-/") 'w3m-google-by-filetype)
#+end_src

(defun global-tags-root-dir ()
  "Return the GTAGS root directory string, or nil if no database exists."
  (with-temp-buffer
    (when (zerop (call-process simple-emacs-plugins-gnu-global-binary nil t nil "-pr"))
      (buffer-substring (point-min) (1- (point-max))))))

(defun global-run-update ()
  "Run `global -u' to update an existing tags database."
  (call-process simple-emacs-plugins-gnu-global-binary nil nil nil "-u"))

(defun global-do-initialize ()
  "Run `gtags' at the top-level .git directory to create a new database."
  (let* ((root (or (upward-find-file ".git") "."))
         (cmd  (concat "cd " root " && " simple-emacs-plugins-gnu-gtags-binary)))
    (shell-command cmd)))

(defun global-run-tags-automatic ()
  "Hook function: update or create the tags database after saving a C/C++ file."
  (when (and simple-emacs-plugins-auto-run-gtags
             (member major-mode '(c++-mode c-mode)))
    (if (global-tags-root-dir)
        (global-run-update)
      (global-do-initialize))))

*** C Mode Hook

Applies the correct coding style for each project:
- Directories in =simple-emacs-linux-cstyle-list= (set in =private.el=) use
  Linux kernel style: 8-space hard tabs, =linux= c-default-style.
- All other directories use 4-space soft tabs.

#+begin_src emacs-lisp
(defun remove-last-dir (dir)
  "Strip the last path component from DIR."
  (let* ((splits (cdr (split-string dir "/")))
         (res    (mapconcat 'identity (butlast splits) "/")))
    (concat "/" res)))

(defun simple-emacs-c-mode-hook ()
  (c-set-offset 'substatement-open 0)
  (c-set-offset 'arglist-intro '+)
  (when simple-emacs-cxx-tabs-as-spaces
    (setq indent-tabs-mode nil))
  ;; Linux kernel style for designated project directories
  (if (and (upward-find-file ".git")
           (member (remove-last-dir (upward-find-file ".git"))
                   simple-emacs-linux-cstyle-list))
      (setq tab-stop-list    '(8 16 24 32 40 48 56 64)
            c-indent-level   8
            c-basic-offset   8
            tab-width        8
            indent-tabs-mode t
            c-default-style  "linux")
    (setq c++-tab-always-indent t
          c-basic-offset        4
          c-indent-level        4
          tab-stop-list         '(4 8 12 16 20 24 28 32 36 40 44 48 52 56 60)
          tab-width             4
          indent-tabs-mode      nil))
  (flymake-mode 1)
  (add-commented-annotations)
  (define-key c-mode-base-map (kbd "C-c C-l")
    (lambda () (interactive) (call-interactively 'compile-next-makefile)))
  (define-key c-mode-base-map (kbd "RET") 'newline-and-indent)
  (define-key c-mode-base-map (kbd "M-.") 'gtags-find-tag)
  (define-key c-mode-base-map (kbd "M-,") 'gtags-find-tag-from-here)

** Multi-Account SMTP

When =simple-emacs-multiple-smtp= is non-nil, an advice on

#+begin_src emacs-lisp

  "Select SMTP settings based on the From: header before sending."
  (when simple-emacs-multiple-smtp
    (save-excursion
      (cl-loop with from = (save-restriction
                              (message-narrow-to-headers)
                              (message-fetch-field "from"))
               for (addr fname server port) in smtp-accounts
               when (or (string-match addr from)
                        (string-match (format "%s <%s>" fname addr) from))
               do (setq user-mail-address      addr
                        user-full-name         fname
                        smtpmail-smtp-server   server
                        smtpmail-smtp-service  port)))))

** Patch Handling

=gnus-article-treat-patch= highlights patch emails with diff-style syntax
coloring.  The helper commands below let you act on a patch without leaving
Gnus:

| Command                                      | Action                                       |
|----------------------------------------------+----------------------------------------------|
| =simple-emacs-gnus-reader-apply-git-patch=   | Apply via =git am= and rebuild               |
| =simple-emacs-gnus-check-patch=              | Run =rhcheckpatch.py= on the patch           |
| =simple-emacs-gnus-write-patch=              | Save the patch to a temp file                |
| =simple-emacs-gnus-view-patch-in-context=    | Open patch in =diff-mode= against tree files |

The tree used by =view-patch-in-context= and =apply-git-patch= is selected
automatically by matching the =[PATCH <prefix>]= subject tag against
=simple-emacs-patch-trees= (set in =private.el=).  If no match is found, both
commands fall back to an interactive directory prompt.

=view-patch-in-context= sets =default-directory= to the tree root before
opening the patch buffer, so =diff-goto-source= (=C-c C-c= or =o= on any
hunk) jumps to the real file at that location — showing the full file
contents rather than just the ±3 lines of diff context.

      (dolist (regex patterns)
        (if (string-match regex line)
            (throw 'done t)))
      (throw 'done nil))))

(defun ft/gnus-atp-looks-like-diff (line)
  "Return `t' if `line' looks remotely like a line from a unified
diff; `nil' otherwise."
  (or (ft/gnus-index-line-p line)
      (ft/gnus-diff-header-p line)
      (ft/gnus-hunk-line-p line)))

;; miscellaneous line handlers
(defun ft/gnus-scissors-line-p (line)
  "Returns `t' if `line' looks like a scissors-line; `nil' otherwise."
  (cond
   ((string-match "^\\( *--* *\\(8<\\|>8\\)\\)+ *-* *$" line) t)
   (t nil)))

;; Patch mail detection
(defvar ft/gnus-article-patch-conditions nil
  "List of conditions that will enable patch treatment.  String
values will be matched as regular expressions within the currently
processed part.  Non-string value are supposed to be code fragments,
which determine whether or not to do treatment: The code needs to
return `t' if treatment is wanted.")

(defun ft/gnus-part-want-patch-treatment ()
  "Run through `ft/gnus-article-patch-conditions' to determine whether
patch treatment is wanted or not. Return `t' or `nil' accordingly."
  (catch 'done
    (dolist (entry ft/gnus-article-patch-conditions)
      (cond
       ((stringp entry)
        (if (re-search-forward entry nil t)
            (throw 'done t)))
       (t
        (if (eval entry)
            (throw 'done t)))))
      (throw 'done nil)))


;; The actual article treatment code
(defun ft/gnus-article-treat-patch-state-machine ()
  "Implement the state machine which colourises a part of an article
if it looks patch-like.

The state machine works like this:

  0a. The machinery starts at the first line of the article's body. Not
      the header lines. We don't care about header lines at all.

  0b. The whole thing works line by line. It doesn't do any forward or
      backward looks.

  1. Initially, we assume, that what we'll see first is part of the
     patch's commit-message. Hence this first initial state is
     \"commit-message\". There are several ways out of this state:

       a) a scissors line is found (see 2.)
       b) a pseudo-header line is found (see 3.)
       c) a three-dashes line is found (see 4.)
       d) something that looks like the start of a unified diff is
          found (see 7.)

  2. A scissors line is something that looks like a pair of scissors running
     through a piece of paper. Like this:

      ------ 8< ----- 8< ------

     or this:

      ------------>8-----------

     The function `ft/gnus-scissors-line-p' decides whether a line is a
     scissors line or not. After a scissors line was treated, the machine
     will switch back to the \"commit-mesage\" state.

  3. This is very similar to a scissors line. It'll just return to the old
     state after its being done. The `ft/gnus-pseudo-header-p' function
     decides if a line is a pseudo header. The line will be appropriately
     coloured.

  4. A three-dashes line is a line that looks like this: \"---\". It's the
     definite end of the \"commit-message\" state. The three dashes line is
     coloured and the state switches to \"commit-comment\". (See 5.)

  5. Nothing in \"commit-comment\" will appear in the generated commit (this
     is git-am specific semantics, but it's useful, so...). It may contain
     things like random comments or - promimently - a diff stat. (See 6.)

  6. A diff stat provides statistics about how much changed in a given commit
     by files and by whole commit (in a summary line). Two functions
     `ft/gnus-diff-stat-line-p' and `ft/gnus-diff-stat-summary-p' decide if a
     line belongs to a diff stat. It's coloured appropriately and the state
     switches back to \"commit-comment\".

  7. There is a function `ft/gnus-unified-diff-line-p' which will cause the
     state to switch to \"unified-diff\" state from either \"commit-message\"
     or \"commit-comment\". In this mode there can be a set of lines types:

       a) diff-header lines (`ft/gnus-diff-header-p')
       b) index lines (`ft/gnus-index-line-p')
       c) hunk lines (`ft/gnus-hunk-line-p')
       d) equals line (\"^==*$\")
       e) context lines (\"^ \")
       f) add lines (\"^\\+\")
       g) remove lines (\"^-\")
       h) empty lines (\"^$\")

     This state runs until the end of the part."
  (catch 'ft/gnus-atp-done
    (let ((state 'commit-message)
          line do-not-move)

      (while t
        ;; Put the current line into an easy-to-handle string variable.
        (setq line
              (buffer-substring-no-properties (point-at-bol) (point-at-eol)))
        (setq do-not-move nil)

        ;; Switched state machine. The "real" states are `commit-message',
        ;; `commit-comment' and `unified-diff'. The other "states" are only
        ;; single-line colourisations that return to their respective parent-
        ;; state. Each state may (throw 'ft/gnus-atp-done) to leave the state-
        ;; machine immediately.
        (setq state
              (cond

               ((eq state 'commit-message)
                (cond
                 ((ft/gnus-scissors-line-p line)
                  (ft/gnus-colour-line 'ft/gnus-scissors)
                  'commit-message)
                 ((ft/gnus-pseudo-header-p line)
                  (ft/gnus-pseudo-header-colour line)
                  'commit-message)
                 ((string= line "---")
                  (ft/gnus-colour-line 'ft/gnus-three-dashes)
                  'commit-comment)
                 ((ft/gnus-atp-looks-like-diff line)
                  (setq do-not-move t)
                  'unified-diff)
                 (t
                  (ft/gnus-colour-line 'ft/gnus-commit-message)
                  'commit-message)))

               ((eq state 'commit-comment)
                (cond
                 ((ft/gnus-diff-stat-line-p line)
                  (ft/gnus-diff-stat-colour line)
                  'commit-comment)
                 ((ft/gnus-diff-stat-summary-p line)
                  (ft/gnus-diff-stat-summary-colour line)
                  'commit-comment)
                 ((ft/gnus-atp-looks-like-diff line)
                  (setq do-not-move t)
                  'unified-diff)
                 (t
                  (ft/gnus-colour-line 'ft/gnus-commit-comment)
                  'commit-comment)))

               ((eq state 'unified-diff)
                (cond
                 ((ft/gnus-diff-header-p line)
                  (ft/gnus-colour-line 'ft/gnus-diff-header)
                  'unified-diff)
                 ((ft/gnus-index-line-p line)
                  (ft/gnus-colour-line 'ft/gnus-diff-index)
                  'unified-diff)
                 ((ft/gnus-hunk-line-p line)
                  (ft/gnus-colour-line 'ft/gnus-diff-hunk)
                  'unified-diff)
                 ((string-match "^==*$" line)
                  (ft/gnus-colour-line 'ft/gnus-diff-equals)
                  'unified-diff)
                 ((string-match "^$" line)
                  'unified-diff)
                 ((string-match "^ " line)
                  (ft/gnus-colour-line 'ft/gnus-diff-context)
                  'unified-diff)
                 ((ft/gnus-atp-misc-diff-p line)
                  (ft/gnus-colour-line 'ft/gnus-diff-misc)
                  'unified-diff)
                 ((string-match "^\\+" line)
                  (ft/gnus-colour-line 'ft/gnus-diff-add)
                  'unified-diff)
                 ((string-match "^-" line)
                  (ft/gnus-colour-line 'ft/gnus-diff-remove)
                  'unified-diff)
                 (t 'unified-diff)))))

        (if (not do-not-move)
            (if (> (forward-line) 0)
                (throw 'ft/gnus-atp-done t)))))))

(defun ft/gnus-article-treat-patch ()
  "Highlight mail parts, that look like patches (well, usually
they *are* patches - or possibly, when you take git's format-patch output,
entire commit exports - including comments).  This treatment assumes the
use of unified diffs.  Here is how it works:

The most fancy type of patch mails look like this:

  From: ...
  Subject: ...
  Other-Headers: ...

  Body text, which can be reflecting the commit message but may
  optionally be followed by a so called scissors line, which
  looks like this (in case of a scissors line, the text above is
  not part of the commit message):

  -------8<----------

  If there really was a scissors line, then it's usually
  followed by repeated mail-headers. Which do not *have* to
  be the same as the one from the sender.

  From: ...
  Subject: ...

  More text. Usually part of the commit message. Likely
  multiline.  What follows may be an optional diffstat. If
  there is one, it's usually preceded by a line that contains
  only three dashes and nothing more. Before the diffstat,
  however, there may be a set of pseudo headers again, like
  these:

  Acked-by: Mike Dev <md@other.tld>
  Signed-off-by: Joe D. User <jdu@example.com>

  ---
  ChangeLog                    |    5 ++++-
  1 file changed, 4 insertions(+), 1 deletions(-)

  Now, there is again room for optional text, which is not
  part of the actual commit message. May be multiline. Actually,
  anything between the three-dashes line and the diff content
  is ignored as far as the commit message goes.

  Now for the actual diff part.  I want this to work for as
  many unified diff formats as possible.  What comes to mind
  is the format used by git and the format used by cvs and
  quilt.

  CVS style looks like this:

  Index: foo/bar.c
  ============================================================
  --- boo.orig/foo/bar.c       2010-02-24 ....
  +++ boo/foo/bar.c            2010-02-28 ....
  @@ -1823,7 +1823,7 @@
  <hunk>

  There may be multiple hunks. Each file gets an \"Index:\" and
  equals line.  Now the git format looks like this:

  diff --git a/ChangeLog b/ChangeLog
  index 6ffbc8c..36e5c17 100644
  --- a/ChangeLog
  +++ b/ChangeLog
  @@ -3,6 +3,9 @@
  <hunk>

  Again, there may be multiple hunks.

  When all hunks and all files are done, there may be additional
  text below the actual text.

And that's it.

You may define the look of several things: pseudo headers, scissor
lines, three-dashes-line, equals lines, diffstat lines, diffstat
summary. Then there is added lines, removed lines, context lines,
diff-header lines and diff-file-header lines, for which we are
borrowing the highlighting faces for from `diff-mode'."
  (if (ft/gnus-part-want-patch-treatment)
      (save-excursion
        (progn
          (let ((inhibit-read-only t))
            (goto-char (point-min))
            (ft/gnus-article-treat-patch-state-machine))))))
#+end_src

Now, fill out our customized version of the patches.

#+begin_src emacs-lisp
  (setq ft/gnus-article-patch-conditions
        '("^@@ -[0-9]+,[0-9]+ \\+[0-9]+,[0-9]+ @@"))

  ;; simple-emacs-patch-trees is populated by private.el.
  (defvar simple-emacs-patch-trees nil
    "Alist mapping patch subject prefixes to git tree configs.
Each entry: (\"prefix\" :path \"/path/to/tree\" :branch \"branchname\").")

  (defun simple-emacs-patch-subject-prefix (subject)
    "Extract the tree prefix from a patch SUBJECT tag.
Returns the first token after PATCH that is not a version (v2) or
fraction (1/3), e.g.:
  \"[PATCH net v2 1/3] foo\" -> \"net\"
  \"[PATCH net-next] foo\"   -> \"net-next\"
  \"[PATCH] foo\"            -> nil"
    (when (string-match "\\[PATCH\\([^]]*\\)\\]" subject)
      (let ((parts (split-string (string-trim (match-string 1 subject)))))
        (cl-find-if (lambda (p)
                      (not (or (string-match-p "^v[0-9]+" p)
                               (string-match-p "^[0-9]+/[0-9]+$" p))))
                    parts))))

  (defun simple-emacs-patch-lookup-tree (subject)
    "Return the tree config plist for the prefix found in SUBJECT, or nil."
    (let ((prefix (simple-emacs-patch-subject-prefix subject)))
      (when prefix
        (cdr (assoc prefix simple-emacs-patch-trees)))))

  (defun simple-emacs-patch-resolve-directory ()
    "Return the git tree directory for the current Gnus article.
Parses the subject for a [PATCH <prefix>] tag, looks it up in
`simple-emacs-patch-trees', and falls back to an interactive prompt
if no match is found."
    (let* ((subject (mail-header-subject
                     (gnus-summary-article-header)))
           (tree    (simple-emacs-patch-lookup-tree subject))
           (path    (when tree (plist-get tree :path))))
      (if path
          (file-name-as-directory (expand-file-name path))
        (read-directory-name "Git tree root: " "~/git/"))))

  (defun simple-emacs-gnus-view-patch-in-context ()
    "Open the current Gnus patch in diff-mode against its configured git tree.
The tree is selected by matching the [PATCH <prefix>] subject tag against
`simple-emacs-patch-trees'.  With the tree root as `default-directory',
\\[diff-goto-source] on any hunk (C-c C-c or o) jumps to the real file,
showing full file context rather than just the diff lines."
    (interactive)
    (let* ((tree-dir       (simple-emacs-patch-resolve-directory))
           (temp-patch-file (make-temp-file "gnus-patch-" nil ".patch")))
      (save-window-excursion
        (gnus-summary-select-article-buffer)
        (write-region (point-min) (point-max) temp-patch-file t))
      (let ((buf (find-file-noselect temp-patch-file)))
        (with-current-buffer buf
          (setq default-directory tree-dir)
          (diff-mode)
          (setq buffer-read-only t))
        (switch-to-buffer buf)
        (message "Tree: %s  —  use C-c C-c or o on a hunk to view in context"
                 tree-dir))))

  (defun simple-emacs-gnus-reader-apply-git-patch ()
    "Apply the current Gnus article as a git patch and run a rebuild command."
    (interactive)
    (let ((temp-patch-file  (make-temp-file "patch"))
          (default-directory (simple-emacs-patch-resolve-directory))
          (build-command     (read-string "Rebuild Command: " "make clean && make")))
      (save-window-excursion
        (gnus-summary-select-article-buffer)
        (write-region (point-min) (point-max) temp-patch-file t))
      (call-process "git" nil "*GNUS Git Patch*" t "am" temp-patch-file)
      (delete-file temp-patch-file)
      (start-process "git-patch" "*GNUS Git Patch*" "/bin/sh" "-c" build-command)
      (switch-to-buffer "*GNUS Git Patch*")))

  (defun simple-emacs-gnus-check-patch ()
    "Run checkpatch.py on the current Gnus article."
    (interactive)
    (let ((temp-patch-file  (make-temp-file "patch"))
          (default-directory (read-string "Target Git Directory: " "~/git/")))
      (save-window-excursion
        (gnus-summary-select-article-buffer)
        (write-region (point-min) (point-max) temp-patch-file t))
      (start-process "git-patch" "*GNUS Git Patch*"
                     "/usr/bin/checkpatch.py" "-v" "-p" "-g"
                     default-directory "-a" temp-patch-file)
      (switch-to-buffer "*GNUS Git Patch*")))

  (defun simple-emacs-gnus-write-patch ()
    "Save the current Gnus article to a temp file and report its path."
    (interactive)
    (let ((temp-patch-file (make-temp-file "patch")))
      (save-window-excursion
        (gnus-summary-select-article-buffer)
        (write-region (point-min) (point-max) temp-patch-file t))
      (message "Patch written to: %s" temp-patch-file)))
